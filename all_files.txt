// CreateStatistics РІС‹С‡РёСЃР»СЏРµС‚ СЃС‚Р°С‚РёСЃС‚РёРєСѓ РёР· РЅР°Р±РѕСЂР° Р·Р°РґР°С‡.
func CreateStatistics(tasks []Task) Statistics {
	if len(tasks) == 0 {
		return NewStatistics(0, 0, nil, nil)
	}

	tasksCreated := len(tasks)

	tasksCompleted := 0
	var totalCompletionDuration time.Duration
	for _, task := range tasks {
		if task.Completed {
			tasksCompleted++

// Р”РѕР±Р°РІР»СЏРµС‚ Р·Р°РіРѕР»РѕРІРєРё Access-Control-Allow-* С‚РѕР»СЊРєРѕ РґР»СЏ СЂР°Р·СЂРµС€С‘РЅРЅС‹С… origins.

// РСЃРїРѕР»СЊР·СѓРµРј map РґР»СЏ O(1) РїРѕРёСЃРєР° РІРјРµСЃС‚Рѕ O(n) РїРµСЂРµР±РѕСЂР° СЃРїРёСЃРєР°.
// Preflight-Р·Р°РїСЂРѕСЃС‹ (OPTIONS) РѕР±СЂР°Р±Р°С‚С‹РІР°СЋС‚СЃСЏ СЃСЂР°Р·Сѓ Р±РµР· РїРµСЂРµРґР°С‡Рё РґР°Р»СЊС€Рµ.
func CORS(allowedOriginsList []string) Middleware {
	allowedOrigins := make(map[string]struct{})
	for _, origin := range allowedOriginsList {
		allowedOrigins[origin] = struct{}{}
	}

	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			origin := r.Header.Get("Origin")

			if _, ok := allowedOrigins[origin]; ok {
				w.Header().Set("Access-Control-Allow-Origin", origin)
				w.Header().Set("Access-Control-Allow-Methods", "GET, POST, PATCH, DELETE, OPTIONS")
				w.Header().Set("Access-Control-Allow-Headers", "Content-Type, Authorization")
			}

	}

type Middleware func(http.Handler) http.Handler

func ChainMiddleware(
	h http.Handler,
	m ...Middleware,
) http.Handler {
	if len(m) == 0 {
		return h
	}

	for i := len(m) - 1; i >= 0; i-- {
		h = m[i](h)
	}
	return h
}

    }

    return nil

    "encoding/json"

    "github.com/cunofou/golang_todoapp/internal/core/errors"

)

type HTTPResponseHandler struct {
    log *logger.Logger
    rw  http.ResponseWriter
}

func NewHTTPResponseHandler(
    log *logger.Logger,
    rw http.ResponseWriter,
) *HTTPResponseHandler {
    return &HTTPResponseHandler{
        log: log,
        rw:  rw,
    }
}

func (h *HTTPResponseHandler) JSONResponse(
    responseBody any,
    statusCode int,
) {
    h.rw.WriteHeader(statusCode)

    if err := json.NewEncoder(h.rw).Encode(responseBody); err != nil {
        h.log.Error("write HTTP response", zap.Error(err))
    }
}

func (h *HTTPResponseHandler) NoContentResponse() {
    h.rw.WriteHeader(http.StatusNoContent)
}

func (h *HTTPResponseHandler) HTMLResponse(htmlFile domain.File) {
    h.rw.WriteHeader(http.StatusOK)

    h.rw.Header().Set("Content-Type", "text/html; charset=utf-8")
    if _, err := h.rw.Write(htmlFile.Buffer()); err != nil {
        h.log.Error("write HTML HTTP response", zap.Error(err))
    }
}

func (h *HTTPResponseHandler) ErrorResponse(err error, msg string) {
    var (
        statusCode int
        logFunc    func(string, ...zap.Field)
    )

    switch {
    case errors.Is(err, errors.ErrInvalidArgument):
        statusCode = http.StatusBadRequest
        logFunc = h.log.Warn

    case errors.Is(err, errors.ErrNotFound):
        statusCode = http.StatusNotFound
        logFunc = h.log.Debug

    case errors.Is(err, errors.ErrConflict):
        statusCode = http.StatusConflict
        logFunc = h.log.Warn

    default:
        statusCode = http.StatusInternalServerError
        logFunc = h.log.Error
    }

    logFunc(msg, zap.Error(err))

    h.errorResponse(
        statusCode,
        err,
        msg,
    )
}

func (h *HTTPResponseHandler) PanicResponse(p any, msg string) {
    statusCode := http.StatusInternalServerError
    err := fmt.Errorf("unexpected panic: %v", p)

    h.log.Error(msg, zap.Error(err))

    h.errorResponse(
        statusCode,
        err,
        msg,
    )
}

func (h *HTTPResponseHandler) errorResponse(
    statusCode int,
    err error,
    msg string,
) {
    response := ErrorResponse{
        Error:   err.Error(),
        Message: msg,
    }

    h.JSONResponse(
        response,
        statusCode,
    )
}

func (r *APIVersionRouter) Handlers() map[string]http.Handler {
	handlers := make(map[string]http.Handler, len(r.routes))

	for _, route := range r.routes {
		// Р¤РѕСЂРјРёСЂСѓРµРј РїРѕР»РЅС‹Р№ РїР°С‚С‚РµСЂРЅ: "GET /api/v1/tasks", "POST /api/v1/users" Рё С‚.Рґ.

		handler := core_http_middleware.ChainMiddleware(
			route.WithMiddleware(),
			r.middleware...,
		)

		handlers[pattern] = handler
	}

	return handlers
}

// Graceful shutdown:

func (s *HTTPServer) Run(ctx context.Context) error {
	// РџСЂРёРјРµРЅСЏРµРј РіР»РѕР±Р°Р»СЊРЅС‹Рµ middleware Рє ServeMux.
	mux := core_http_middleware.ChainMiddleware(s.mux, s.middleware...)

	server := &http.Server{
		Addr:    s.config.Addr,
		Handler: mux,
	}

	// Р‘СѓС„РµСЂРёР·РёСЂРѕРІР°РЅРЅС‹Р№ РєР°РЅР°Р» (СЂР°Р·РјРµСЂ 1), С‡С‚РѕР±С‹ РіРѕСЂСѓС‚РёРЅР° РЅРµ Р·Р°Р±Р»РѕРєРёСЂРѕРІР°Р»Р°СЃСЊ
	// РїСЂРё РѕС‚РїСЂР°РІРєРµ РѕС€РёР±РєРё, РµСЃР»Рё РѕСЃРЅРѕРІРЅРѕР№ РїРѕС‚РѕРє СѓР¶Рµ СѓС€С‘Р» РІ select.
	ch := make(chan error, 1)

	go func() {
		defer close(ch)

		s.log.Warn("start HTTP server", zap.String("addr", s.config.Addr))

		err := server.ListenAndServe()

		// http.ErrServerClosed вЂ” РЅРѕСЂРјР°Р»СЊРЅРѕРµ Р·Р°РІРµСЂС€РµРЅРёРµ РїРѕСЃР»Рµ Shutdown(), РЅРµ РѕС€РёР±РєР°.
		if !errors.Is(err, http.ErrServerClosed) {
			ch <- err
		}
	}()

	select {
	case err := <-ch:
		// HTTP СЃРµСЂРІРµСЂ Р·Р°РІРµСЂС€РёР»СЃСЏ РёР·-Р·Р° РѕС€РёР±РєРё (РЅР°РїСЂРёРјРµСЂ, РїРѕСЂС‚ Р·Р°РЅСЏС‚).
		if err != nil {
			return fmt.Errorf("listen and server HTTP: %w", err)
		}
	case <-ctx.Done():
		// РџРѕР»СѓС‡РµРЅ СЃРёРіРЅР°Р» Р·Р°РІРµСЂС€РµРЅРёСЏ вЂ” РІС‹РїРѕР»РЅСЏРµРј graceful shutdown.
		s.log.Warn("shutdown HTTP server...")

		shutdownCtx, cancel := context.WithTimeout(
			context.Background(),
			s.config.ShutdownTimeout,
		)
		defer cancel()

		if err := server.Shutdown(shutdownCtx); err != nil {
			// Р•СЃР»Рё graceful shutdown РЅРµ СѓСЃРїРµР» вЂ” РїСЂРёРЅСѓРґРёС‚РµР»СЊРЅРѕ Р·Р°РєСЂС‹РІР°РµРј.
			_ = server.Close()

			return fmt.Errorf("shutdown HTTP server: %w", err)
		}

		s.log.Warn("HTTP server stopped")
	}

	return nil
}

// WHERE-СѓСЃР»РѕРІРёСЏ СЃ РїСЂРѕРёР·РІРѕР»СЊРЅС‹Рј С‡РёСЃР»РѕРј С„РёР»СЊС‚СЂРѕРІ (userID, from, to).
// РџР°СЂР°РјРµС‚СЂС‹ РґРѕР±Р°РІР»СЏСЋС‚СЃСЏ РІ args РґРёРЅР°РјРёС‡РµСЃРєРё: $1, $2, ... вЂ” СЌС‚Рѕ Р·Р°С‰РёС‰Р°РµС‚ РѕС‚ SQL-РёРЅСЉРµРєС†РёР№.
func (r *StatisticsRepository) GetTasks(
	ctx context.Context,
	userID *uuid.UUID,
	from *time.Time,
	to *time.Time,
) ([]domain.Task, error) {
	ctx, cancel := context.WithTimeout(ctx, r.pool.OpTimeout())
	defer cancel()

	var queryBuilder strings.Builder

	queryBuilder.WriteString(`
	SELECT id, version, title, description, completed, created_at, completed_at, author_user_id
	FROM todoapp.tasks
	`)

	args := []any{}
	conditions := []string{}

	if userID != nil {
		conditions = append(conditions, fmt.Sprintf("author_user_id=$%d", len(args)+1))
		args = append(args, userID)
	}

	if from != nil {
		conditions = append(conditions, fmt.Sprintf("created_at>=$%d", len(args)+1))
		args = append(args, from)
	}

	if to != nil {
		conditions = append(conditions, fmt.Sprintf("created_at<$%d", len(args)+1))
		args = append(args, to)
	}

	if len(conditions) > 0 {
		queryBuilder.WriteString(" WHERE " + strings.Join(conditions, " AND "))
	}

	queryBuilder.WriteString(" ORDER BY id ASC")

	rows, err := r.pool.Query(ctx, queryBuilder.String(), args...)
	if err != nil {
		return nil, fmt.Errorf("select tasks: %w", err)
	}
	defer rows.Close()

	var taskModels []TaskModel

	for rows.Next() {
		var taskModel TaskModel
		if err := taskModel.Scan(rows); err != nil {
			return nil, fmt.Errorf("scan tasks: %w", err)
		}

		taskModels = append(taskModels, taskModel)
	}
	if err := rows.Err(); err != nil {
		return nil, fmt.Errorf("next rows: %w", err)
	}

	taskDomains := modelsToDomains(taskModels)

	return taskDomains, nil
}

// GetTask РІРѕР·РІСЂР°С‰Р°РµС‚ Р·Р°РґР°С‡Сѓ РїРѕ ID.
// Р•СЃР»Рё Р·Р°РґР°С‡Р° РЅРµ РЅР°Р№РґРµРЅР° вЂ” ErrNoRows РёР· РїСѓР»Р° С‚СЂР°РЅСЃР»РёСЂСѓРµС‚СЃСЏ РІ core_errors.ErrNotFound,
// С‡С‚Рѕ РїСЂРёРІРµРґС‘С‚ Рє HTTP 404 РІ С‚СЂР°РЅСЃРїРѕСЂС‚РЅРѕРј СЃР»РѕРµ.
func (r *TasksRepository) GetTask(
	ctx context.Context,
	id uuid.UUID,
) (domain.Task, error) {
	ctx, cancel := context.WithTimeout(ctx, r.pool.OpTimeout())
	defer cancel()

	query := `
	SELECT id, version, title, description, completed, created_at, completed_at, author_user_id
	FROM todoapp.tasks
	WHERE id=$1;
	`

	row := r.pool.QueryRow(ctx, query, id)

	var taskModel TaskModel
	if err := taskModel.Scan(row); err != nil {
		if errors.Is(err, core_postgres_pool.ErrNoRows) {
			return domain.Task{}, fmt.Errorf(
				"task with id='%s': %w",
				id,
				core_errors.ErrNotFound,
			)
		}

		return domain.Task{}, fmt.Errorf("scan error: %w", err)

func (r *TasksRepository) GetTasks(
	ctx context.Context,
	userID *uuid.UUID,
	limit *int,
	offset *int,
) ([]domain.Task, error) {
	ctx, cancel := context.WithTimeout(ctx, r.pool.OpTimeout())
	defer cancel()

	query := `
	SELECT id, version, title, description, completed, created_at, completed_at, author_user_id
	FROM todoapp.tasks
	%s
	ORDER BY id ASC
	LIMIT $1
	OFFSET $2;
	`

	args := []any{limit, offset}

	if userID != nil {
		query = fmt.Sprintf(query, "WHERE author_user_id=$3")
		args = append(args, userID)
	} else {
		query = fmt.Sprintf(query, "")
	}

	rows, err := r.pool.Query(
		ctx,
		query,
		args...,
	)
	if err != nil {
		return nil, fmt.Errorf("select tasks: %w", err)
	}
	defer rows.Close()

	var taskModels []TaskModel

	for rows.Next() {
		var taskModel TaskModel

		if err := taskModel.Scan(rows); err != nil {
			return nil, fmt.Errorf("scan tasks: %w", err)
		}

		taskModels = append(taskModels, taskModel)
	}
	if err := rows.Err(); err != nil {
		return nil, fmt.Errorf("next rows: %w", err)
	}

	taskDomains := modelsToDomains(taskModels)

	return taskDomains, nil
}

	if err != nil {
		return domain.Task{}, fmt.Errorf("get task from repository: %w", err)
	}

	if err := task.ApplyPatch(patch); err != nil {
		return domain.Task{}, fmt.Errorf("apply task patch: %w", err)
	}

	patchedTask, err := s.tasksRepository.UpdateTask(ctx, task)
	if err != nil {

func (r *UsersRepository) UpdateUser(
	ctx context.Context,
	user domain.User,
) (domain.User, error) {
	ctx, cancel := context.WithTimeout(ctx, r.pool.OpTimeout())
	defer cancel()

	query := `
	UPDATE todoapp.users
	SET
		full_name=$1,
		phone_number=$2,
		version=version+1
	WHERE id=$3 AND version=$4
	RETURNING
		id,
		version,
		full_name,
		phone_number;
	`

	row := r.pool.QueryRow(
		ctx,
		query,
		user.FullName,
		user.PhoneNumber,
		user.ID,
		user.Version,
	)

	var userModel UserModel
	err := row.Scan(
		&userModel.ID,
		&userModel.Version,
		&userModel.FullName,
		&userModel.PhoneNumber,
	)
	if err != nil {
		if errors.Is(err, core_postgres_pool.ErrNoRows) {
			return domain.User{}, fmt.Errorf(
				"user with id='%s' concurrently accessed: %w",
				user.ID,
				core_errors.ErrConflict,
			)
		}

		return domain.User{}, fmt.Errorf("scan error: %w", err)
	}

	userDomain := modelToDomain(userModel)

	return userDomain, nil
}
